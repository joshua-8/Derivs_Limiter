635 float tempVelLimit;
636 if (accelLimit == INFINITY && decelLimit == INFINITY)
637 tempVelLimit = _dist / _time;
638 else {

641 acc = decelLimit * 2;

643 acc = accelLimit * 2;

726 virtual float _calc()
727 {
728 if (positionPointer && !isnan(*positionPointer))
729 position = *positionPointer;
730
731 if (velocityPointer && !isnan(*velocityPointer))
732 velocity = *velocityPointer;
733
734 time = (micros() - lastTime) / 1000000.0;
735 if (lastTime == 0) {
736 time = 0; // in case there's a delay between starting the program and the first calculation avoid jump at start
737 lastTime = micros();
738 }
739 if (time == 0) {
740 return position;
741 }
742 lastTime = micros();
743
744 // constrain positions within limits
745 if (position > posLimitHigh) {
746 position = posLimitHigh;
747 velocity = 0;
748 } else if (position < posLimitLow) {
749 position = posLimitLow;
750 velocity = 0;
751 }
752 target = constrain(target, posLimitLow, posLimitHigh);
753
754 targetDelta = target - lastTarget;
755 lastTarget = target;
756
757 if (preventGoingTooFast) {
758 velocity = constrain(velocity, -velLimit, velLimit);
759 }
760 if (posMode) {
761 if (preventGoingWrongWay && velocity != 0 && target != position && ((velocity > 0) != (target - position > 0))) { // going the wrong way
762 velocity = 0;
763 }
764
765 if (velocity == 0 && position == target) { // if stopped at the target, no calculations are needed
766 accel = 0;
767 return position;
768 }
769
770 if (velocity != 0 && target != position && (velocity > 0) == (target - position > 0)
771 && (abs(position - target) - abs(velocity * (time)) <= sq(velocity) / 2.0 / decelLimit)) {
772 // predicted to be too close next time, decel now.
773 if (abs(position - target) <= abs(velocity * time) && (abs(velocity) <= decelLimit * maxStoppingDecel * time)) { // close enough and slow enough, just stop
774 accel = 0;
775 velocity = 0;
776 position = target;
777 } else { // decel
778 accel = -sq(velocity) / 2.0 / (target - (position));
779 accel = constrain(accel, -decelLimit * maxStoppingDecel, decelLimit * maxStoppingDecel);
780 velocity += accel * time;
781 position += velocity * time;
782 }
783 } else if (velocity != 0 && target != position && (velocity > 0) != (target - position > 0)) { // if going wrong way, decel
784 accel = ((target - position > 0) ? decelLimit : -decelLimit);
785 velocity += accel * time;
786 if (velocity != 0 && (velocity > 0) == (target - position > 0)) { // switched direction, stop at zero velocity, in case accel is lower
787 velocity = 0;
788 accel = 0;
789 } else {
790 position += velocity * time;
791 }
792 } else if (abs(velocity) < velLimit) { // too slow, speed up
793 float tempVelocity = velocity;
794 accel = (position > target) ? -accelLimit : accelLimit;
795 velocity += accel * time;
796 velocity = constrain(velocity, -velLimit, velLimit);
797 float maxSpeedThatCanBeStopped = sqrt(2 * (decelLimit)*abs(position - target)); // v^2 = u^2 + 2as
798 velocity = constrain(velocity, -maxSpeedThatCanBeStopped, maxSpeedThatCanBeStopped);
799 accel = (velocity - tempVelocity) / time;
800 position += velocity * time;
801 if (abs(position - target) <= abs(velocity * time) && (abs(velocity) <= decelLimit * maxStoppingDecel * time)) { // close enough and slow enough, just stop
802 accel = 0;
803 velocity = 0;
804 position = target;
805 }
806 } else if (abs(velocity) > velLimit) { // too fast, slow down
807 boolean velPositive = (velocity > 0);
808 float tempVelocity = velocity;
809 velocity += velPositive ? -decelLimit * time : decelLimit * time;
810 if (velPositive) {
811 if (velocity < velLimit) {
812 velocity = velLimit;
813 }
814 } else { // vel negative
815 if (velocity > -velLimit) {
816 velocity = -velLimit;
817 }
818 }
819
820 accel = (velocity - tempVelocity) / time;
821 position += velocity * time;
822 } else { // coast, no accel
823 accel = 0;
824 position += velocity * time;
825 }
826 } else { // not pos mode, vel mode
827 float tempVelocity = velocity;
828 velocityTarget = constrain(velocityTarget, -velLimit, velLimit);
829 if (preventGoingWrongWay && velocity != 0 && velocityTarget != 0 && (velocity > 0) != (velocityTarget > 0)) {
830 velocity = 0;
831 }
832 if (velocity != velocityTarget) {
833 if (velocityTarget == 0) {
834 velocity += constrain(velocityTarget - velocity, -decelLimit * time, decelLimit * time);
835 } else if (velocityTarget > 0) {
836 velocity += constrain(velocityTarget - velocity, -decelLimit * time, accelLimit * time);
837 if (velocity < 0) { // prevent decel from crossing zero and causing accel
838 velocity = 0;
839 }
840 } else {
841 velocity += constrain(velocityTarget - velocity, -accelLimit * time, decelLimit * time);
842 if (velocity > 0) { // prevent decel from crossing zero and causing accel
843 velocity = 0;
844 }
845 }
846 }
847 accel = (velocity - tempVelocity) / time;
848 position += velocity * time;
849 }
850
851 if (positionPointer)
852 *positionPointer = position;
853 if (velocityPointer)
854 *velocityPointer = velocity;
855
856 posDelta = position - lastPos;
857 lastPos = position;
858
859 return position;
860 }
